    Function regex

    convenient function to instantiate a regexparser

    Parser!T regex(T) (
      T[] s,
      bool collect = true
    );

    Example

    regexParser 

    import unit_threaded;

auto res = regex("(a)(.)(c)").parse("abcd");
res.success.shouldBeTrue;
res.results.shouldEqual(["abc", "a", "b", "c"]);
res.rest.shouldEqual("d");

    import unit_threaded;
